from pydantic import BaseModel, constr, Field
from uuid import UUID

class RegenerateRequest(BaseModel):
    user_id: UUID
    memo_id: UUID
    section: constr(strip_whitespace=True, min_length=1)
    user_prompt: str | None = Field(None, alias="user prompt")

    class Config:
        allow_population_by_field_name = True  # accepts both "user_prompt" and "user prompt"
