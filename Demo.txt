import pandas as pd
from name_matching.name_matcher import NameMatcher

# Sample vendor lists
df_master = pd.DataFrame({
    'Company name': [
        'Tredence',
        'Tredence Analytics Solutions Private Limited',
        'Tredent Solutions',
        'Infosys Ltd',
        'Infosys Limited',
        'ABC Technologies',
        'XYZ Corporation'
    ]
})

df_to_match = pd.DataFrame({
    'Company name': [
        'Tredence Analytics Pvt Ltd',
        'Inforsys Limited',  # intentional typo
        'ABC Tech',
        'X Y Z Corp'
    ]
})

# Initialize the matcher
matcher = NameMatcher(
    number_of_matches=3,     # how many matches per query
    legal_suffixes=True,      # try to handle legal suffixes
    common_words=False,
    top_n=50,
    verbose=True
)

# Set the distance metrics to use
matcher.set_distance_metrics(
    discounted_levenshtein=True,
    bag=True,
    typo=True,
    refined_soundex=True
)

# Load the master data that we want to match against
matcher.load_and_process_master_data(
    df_matching_data=df_master,
    column='Company name',
    transform=True
)

# Perform matching
results = matcher.match_names(
    to_be_matched=df_to_match,
    column_matching='Company name'
)

print("\nMatching Results:")
print(results)
