def categorize_attendees(attendees_list):
    """
    Categorize attendees based on the company name.
    - NT Attendees: If company contains 'Northern Trust'
    - Consultant Attendees: If company contains 'Consult'
    - Client Attendees: All others
    """
    client_attendees = []
    nt_attendees = []
    consultant_attendees = []

    for attendee in attendees_list:
        company = attendee.get("company", "").lower()
        if "northern trust" in company:
            nt_attendees.append(attendee)
        elif "consult" in company:
            consultant_attendees.append(attendee)
        else:
            client_attendees.append(attendee)

    return client_attendees, nt_attendees, consultant_attendees

# Example input JSON
input_json = {
    "nt": [
        {
            "name": "Patricia Anderson",
            "title": "Managing Director",
            "company": "Northern Trust"
        },
        {
            "name": "Rachel Kim",
            "title": "Portfolio Manager, Fixed Income",
            "company": "Northern Trust"
        },
        {
            "name": "Jase Auby",
            "title": "Chief Investment Officer",
            "company": "Texas Teachers Retirement System"
        },
        {
            "name": "Jerry Albright",
            "title": "Chief Financial Officer",
            "company": "Texas Teachers Retirement System"
        },
        {
            "name": "Steven Foresti",
            "title": "Senior Consultant",
            "company": "Aon Investment Consulting"
        }
    ]
}

# Extract attendees and categorize them
all_attendees = input_json.get("nt", [])
client_attendees, nt_attendees, consultant_attendees = categorize_attendees(all_attendees)

# Optional: Print or return the result
result = {
    "client_attendees": client_attendees,
    "nt_attendees": nt_attendees,
    "consultant_attendees": consultant_attendees
}

import json
