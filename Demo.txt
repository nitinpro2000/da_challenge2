import json

def categorize_contacts(contact_data):
    clients = []
    consultants = []

    for contact in contact_data:
        name = contact.get("contact_name")
        ctype = contact.get("contact_type")

        if ctype == "Client":
            clients.append(name)
        elif ctype == "Consultant":
            consultants.append(name)

    result = {
        "client_contacts": clients,
        "consultant_contacts": consultants
    }

    return json.dumps(result, indent=2)
