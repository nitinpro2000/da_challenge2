from name_matching import NameMatcher

vendors = [
    "Tredence",
    "Tredence Analytics Solutions Private Limited",
    "Tredent Solutions",
    "Infosys Ltd",
    "Infosys Limited",
    "ABC Technologies",
    "XYZ Corporation"
]

# Initialize matcher
matcher = NameMatcher(n_jobs=1, verbose=False)

# Load vendor database
matcher.load_and_process_master_data(vendors)

# Query name
query = "Tredence"

# Find matches
matches = matcher.match_names([query], top_n=5)

print("Query:", query)
print("\nTop Matches:")
print(matches)
