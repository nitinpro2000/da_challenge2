-- Enable pgvector extension
CREATE EXTENSION IF NOT EXISTS vector;

-- Create the table
CREATE TABLE historical_news_data (
    article_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    company_name TEXT NOT NULL,
    industry_name TEXT,
    type_of_article TEXT,
    article_url TEXT,
    article_search_term TEXT,
    article_title TEXT,
    article_date DATE,
    article_text TEXT,
    article_summary TEXT,
    date_of_extraction DATE DEFAULT CURRENT_DATE,
    relevance_score REAL,
    article_embedding VECTOR(1536),
    article_summary_embedding VECTOR(1536),
    inserted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
