import streamlit as st

# App Title
st.title("Item List Display")

# Items to display
items = [
    "Item 1: Apple",
    "Item 2: Banana",
    "Item 3: Orange",
    "Item 4: Mango",
    "Item 5: Pineapple"
]

# Display items with a styled header
st.header("Available Items")

# Using an expander for better UI
with st.expander("View Items"):
    for item in items:
        st.markdown(f"- **{item}**")

# Option to select an item
selected_item = st.selectbox("Select an item:", items)

# Display the selected item
st.success(f"You selected: {selected_item}")

# Add more interactivity (Optional)
if st.button("Confirm Selection"):
    st.write(f"Confirmed: {selected_item}")
