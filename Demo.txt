import re

def extract_unique_citations(text):
    """
    Extracts unique citation numbers from a given string.
    Only considers numeric citations like [1], [2], etc.

    Args:
        text (str): Input string containing citations.

    Returns:
        List[int]: Sorted list of unique citation numbers.
    """
    citations = re.findall(r'\[(\d+)\]', text)
    return sorted(set(map(int, citations)))

def get_citation_ids_from_text(text, citation_dict):
    """
    Returns citation IDs from a citation dictionary based on citations found in text.

    Args:
        text (str): Input text with citation numbers like [1], [2]
        citation_dict (dict): Mapping from citation number as string to citation ID

    Returns:
        List[str]: List of citation IDs present in the text
    """
    citation_numbers = extract_unique_citations(text)
    citation_ids = [citation_dict[str(num)] for num in citation_numbers if str(num) in citation_dict]
    return citation_ids
